  public static int dfs(int [] v){//dfsë¡œ íƒìƒ‰í•˜ë˜, ë°©ë¬¸í•œ node ê°’ì€ 2ë¡œ ë°”ê¾¸ê¸°
    Stack<int[]> stack = new Stack<>();

      int[] top = stack.peek();
      boolean flag = false;

        }
      }
      if (flag == false)
        stack.pop();
    }
    return result;
  }
